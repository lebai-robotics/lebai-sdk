endif()

file(GLOB_RECURSE CPP_SRCS CONFIGURE_DEPENDS "*.cc")

endforeach()

  file(GLOB_RECURSE DOTNET_SRCS CONFIGURE_DEPENDS "*.cs")
  foreach(FILE_NAME IN LISTS DOTNET_SRCS)
    add_dotnet_example(${FILE_NAME})
  endforeach()
endif()

if(BUILD_JAVA)
  file(GLOB_RECURSE JAVA_SRCS CONFIGURE_DEPENDS "*.java")
  foreach(FILE_NAME IN LISTS JAVA_SRCS)
    add_java_example(${FILE_NAME})
  endforeach()
endif()
